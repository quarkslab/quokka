138 const Reference& get(const Reference& ref) {

141 return **it;

143 throw std::out_of_range("Reference does not exist");

146 // Overload emplace function
147 template <typename... Args>
148 const Reference& emplace(Args&&... args) {
149 const Reference& ref = Base::emplace(std::forward<Args>(args)...);
150 auto [it, end] = this->pending_links.equal_range(ref);
151 while (it != end) {
152 it->second->push_back(&ref);
153 it = this->pending_links.erase(it);
154 }
155 return ref;
156 }
157
158 // Overload insert function
159 template <typename... Args>
160 const Reference& insert(Args&&... args) {
161 const Reference& ref = Base::insert(std::forward<Args>(args)...);
162 auto [it, end] = this->pending_links.equal_range(ref);
163 while (it != end) {
164 it->second->push_back(&ref);
165 it = this->pending_links.erase(it);
166 }
167 return ref;
168 }
169
170 void assert_no_pending_link() const { assert(this->pending_links.empty()); }
171
181 void attach_link(std::vector<const Reference*>* ref_vector,
182 const Reference& ref) {
183 if (this->contains(ref))
184 ref_vector->push_back(std::addressof(this->get(ref)));
185 else
186 this->pending_links.insert({ref, ref_vector});
187 }
