45// Utility class that offers a view over a sorted collection of objects
46template <typename Storage, typename T>
47 requires requires(Storage const& s) {
48 { s.sorted_view } -> std::same_as<const std::vector<const T*>&>;
49 }
50class SortedViewImpl {
51 public:
52 explicit SortedViewImpl(std::shared_ptr<const Storage> storage)
53 : storage(std::move(storage)),
54 deref_view(this->storage->sorted_view |
55 std::views::transform(deref_func)) {
56 if (!this->storage)
57 throw std::logic_error("SortedViewImpl cannot have a null storage");
58 }
59
60 /* Iterators proxy */
61 auto begin() const { return deref_view.begin(); }
62 auto end() const { return deref_view.end(); }
63
64 size_t size() const { return storage->sorted_view.size(); }
65
66 const T& operator[](size_t index) const {
67 if (index >= storage->sorted_view.size())
68 throw std::out_of_range("Index out of range");
69 return *storage->sorted_view[index];
70 }
71
72 private:
73 static constexpr auto deref_func = [](const T* p) -> const T& { return *p; };
74
75 std::shared_ptr<const Storage> storage;
76 decltype(storage->sorted_view | std::views::transform(deref_func)) deref_view;
77};

145// Concept that describes a sorted view. The view offers standard iterator
146// access, random access and the collection size
147template <typename View, typename T>
148concept SortedViewT = requires(const View& v, size_t i) {
149 { v.begin() } -> std::input_iterator;
150 { v.end() } -> std::sentinel_for<decltype(v.begin())>;
151 { v.size() } -> std::convertible_to<size_t>;
152 { *v.begin() } -> std::same_as<const T&>;
153 { v[i] } -> std::same_as<const T&>;
154};

156// Concept that describes a Storage used for a bucket
157template <typename StorageT, typename T>
158concept StorageLike = requires(const StorageT& storage) {
159 { storage.bucket.size() } -> std::convertible_to<size_t>;
160 { storage.bucket_values() } -> std::ranges::input_range;
161 requires std::same_as<std::remove_cvref_t<decltype(storage.sorted_view)>,
162 std::vector<const T*>>;
163 requires std::convertible_to<
164 std::ranges::range_reference_t<decltype(storage.bucket_values())>,
165 const T&>;
166};

168template <std::derived_from<ProtoHelper> P, StorageLike<P> StorageT>
169class CommonSortableBucket {
170 private:
171 bool sorted = false;
172
173 protected:
177 std::shared_ptr<StorageT> storage = std::make_shared<StorageT>();
178
179 bool frozen = false;
180
181 public:
182 using SortedView = detail::SortedViewImpl<StorageT, P>;
183
188 size_t size() const { return storage->bucket.size(); }
189
195 void Freeze() {
196 if (this->frozen)
197 return;
198
199 this->frozen = true;
200 }
201
202 void Sort() {
203 if (this->sorted)
204 return;
205
206 if (!this->frozen)
207 this->Freeze();
208
209 storage->sorted_view.clear();
210 storage->sorted_view.reserve(this->size());
211 for (const auto& e : storage->bucket_values())
212 storage->sorted_view.push_back(&e);
213
214 std::sort(
215 storage->sorted_view.begin(), storage->sorted_view.end(),
216 [](const P* a, const P* b) { return a->ref_count > b->ref_count; });
217 this->sorted = true;
218 }
219
228 SortedView GetSortedView() const {
229 if (!this->frozen)
230 throw std::logic_error(
231 "Bucket must be frozen before getting the sorted view");
232 if (!this->sorted)
233 throw std::logic_error(
234 "Bucket must be sorted before getting the sorted view");
235
236 return SortedView(storage);
237 }
238};

322template <typename K, std::derived_from<ProtoHelper> P,
323 typename Compare = std::less<K>>
324// TODO add hashable and comparable type
325class MapBucket
326 : public CommonSortableBucket<P, detail::MapStorage<K, P, Compare>> {
327 public:
338 template <typename... Args>
339 const P& emplace(const K& key, Args&&... args) {
340 if (this->frozen)
341 throw std::logic_error("Cannot insert new elements in a frozen bucket");
342
343 // Search if it's already in the bucket
344 if (auto search = this->storage->bucket.find(key);
345 search != this->storage->bucket.end()) {
346 ++search->second->ref_count;
347 return *search->second;
348 }
349
350 // Otherwise insert it
351 auto [it, ok] = this->storage->bucket.emplace(
352 key, std::make_unique<P>(std::forward<Args>(args)...));
353 it->second->ref_count++;
354 return *it->second;
355 }
356
363 bool contains(const K& key) const {
364 return this->storage->bucket.contains(key);
365 }
366
367 const P& operator[](const K& key) const {
368 if (!this->contains(key))
369 throw std::out_of_range("Key not in the collection");
370 return *this->storage->bucket.at(key);
371 }
372};
