190struct Position {
191 PositionType pos_type;
192 int64 x;
193 int64 y;
194
199 bool operator<(const Position& pos) const {
200 return pow(x + y, 2) < pow(pos.x + pos.y, 2);
201 };
202
210 template <typename H>
211 friend H AbslHashValue(H h, const Position& m) {
212 return H::combine(std::move(h), m.x, m.y, m.pos_type);
213 }
214
218 bool operator==(const Position& pos) const {
219 return pos.x == x && pos.y == y && pos_type == pos.pos_type;
220 }
