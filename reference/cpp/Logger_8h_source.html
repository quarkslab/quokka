171 void WriteLine(const Record& record) {
172 std::string message = (m_level == LogLevel::DEBUG) ? record.GetFullMessage()
173 : record.GetMessage();
174
175 std::string formatted =
176 absl::StrCat(LogLevelToString(record.GetLevel()), " ",
177 absl::FormatTime("%H:%M:%S", absl::Now(), this->m_tz), " ",
178 message, "\n");
179
180 fputs(formatted.c_str(), stderr);
181 fflush(stderr);
182
183 if (m_defaultui) {
184 // TODO(dm) fix me because this is not available when running tests
185 msg("%s", formatted.c_str());
186 }
187 }
188
192 void Flush() {
193 fflush(stderr);
194 fflush(stdout);
195 if (m_defaultui) {
196 // Force IDA to flush its message buffer
197 refresh_idaview_anyway();
198 }
199 }

205 static Logger& GetInstance() {
206 static Logger instance;
207 return instance;
208 }

214 void operator+=(const Record& record) {
215 if (CheckLevel(record.GetLevel())) {
216 WriteLine(record);
