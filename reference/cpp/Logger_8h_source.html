127 static const char* LogLevelToString(LogLevel level) {
128 switch (level) {
129 case DEBUG:
130 return "DEBUG";
131 case WARNING:
132 return "WARNING";
133 case INFO:
134 return "INFO";
135 case ERROR:
136 return "ERROR";
137 case FATAL:
138 return "FATAL";

141 return "UNKNOWN";
142 }

166 void WriteLine(const Record& record) {
167 std::string message = (m_level == LogLevel::DEBUG) ? record.GetFullMessage()
168 : record.GetMessage();
169
170 std::string formatted =
171 absl::StrCat(LogLevelToString(record.GetLevel()), " ",
172 absl::FormatTime("%H:%M:%S", absl::Now(), this->m_tz), " ",
173 message, "\n");
174
175 fputs(formatted.c_str(), stderr);
176 fflush(stderr);
177
178 if (m_defaultui) {
179 // TODO(dm) fix me because this is not available when running tests
180 msg("%s", formatted.c_str());
181 }
182 }
183
188 static Logger& GetInstance() {
189 static Logger instance;
190 return instance;
191 }
192
197 void operator+=(const Record& record) {
198 if (CheckLevel(record.GetLevel())) {
199 WriteLine(record);
200 }
201 }
