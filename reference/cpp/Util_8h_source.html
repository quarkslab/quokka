104 public:

117 template <typename... Args>
118 std::shared_ptr<P> emplace(Args&&... args) {
119 static_assert(std::is_base_of<ProtoHelper, P>::value,
120 "P must inherit from ProtoHelper");
121 auto [it, result] =
122 bucket.emplace(std::make_shared<P>(std::forward<Args>(args)...));
123 (*it)->ref_count++;
124 return *it;
125 }
126
127 /* Iterators proxy */
128 iterator begin() { return bucket.begin(); }
129 iterator end() { return bucket.end(); }
130 [[nodiscard]] const_iterator begin() const { return bucket.begin(); }
131 [[nodiscard]] const_iterator end() const { return bucket.end(); }
132
137 [[nodiscard]] size_t size() const { return bucket.size(); }
138
139 using frequency_map =
140 absl::btree_multimap<uint64_t, std::shared_ptr<P>, std::greater<>>;
141
150 [[nodiscard]] frequency_map SortByFrequency() const {
151 frequency_map ordered_map;
152 // ordered_map.reserve(bucket.size());
153 for (auto const element : bucket) {
154 ordered_map.emplace(element->ref_count, element);
155 }
156
157 return ordered_map;
158 }

163 void clear() { this->bucket.clear(); }

170 void erase(const std::shared_ptr<P>& key) { this->bucket.erase(key); }
171};

256template <typename ForwardIterator, typename OutputIterator, typename Equal,
257 typename Merge>
258void MergeAdjacent(ForwardIterator first, ForwardIterator last,
259 OutputIterator out, Equal equal, Merge merge) {
260 for (auto lb = first, ub = last; lb != last; lb = ub)
261 *out++ = std::accumulate(
262 lb + 1, ub = std::mismatch(lb + 1, last, lb, equal).first, *lb, merge);
263}
