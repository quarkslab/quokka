195template <std::invocable F>
196class scope_exit_guard {
197 public:
198 explicit scope_exit_guard(F&& f) noexcept(
199 std::is_nothrow_move_constructible_v<F>)
200 : f_(std::forward<F>(f)) {}
201
202 scope_exit_guard(scope_exit_guard&& other) noexcept(
203 std::is_nothrow_move_constructible_v<F>)
204 : f_(std::move(other.f_)), active_(std::exchange(other.active_, false)) {}
205
206 scope_exit_guard(const scope_exit_guard&) = delete;
207 scope_exit_guard& operator=(const scope_exit_guard&) = delete;
208 scope_exit_guard& operator=(scope_exit_guard&&) = delete;
209
210 void release() noexcept { active_ = false; }
211
212 ~scope_exit_guard() noexcept {
213 if (!active_)
214 return;
215 try {
216 f_();
217 } catch (...) { // never throw from dtors
218 }
219 }
220
221 private:
222 F f_;
223 bool active_ = true;
224};

226[[nodiscard]] inline auto scoped_step(
227 std::string_view start_msg, std::string_view done_msg,
228 std::optional<std::string_view> wait_box = std::nullopt) {
229 Timer timer(absl::Now());
230
231 if (wait_box && !wait_box->empty()) {
232 replace_wait_box("%.*s", static_cast<int>(wait_box->size()),
233 wait_box->data());
234 }
235 QLOGI << start_msg;
236
237 // Own strings so passing temporaries is always safe.
238 std::string done = std::string(done_msg);
239
240 return scope_exit_guard(
241 [timer = std::move(timer), done = std::move(done)]() mutable noexcept {
242 const auto secs = timer.ElapsedSeconds(absl::Now());
243 QLOGI << absl::StrFormat("%s (took: %.2fs)", done, secs);
244 });
245}
