54static void ExportCompositeMembers(
55 std::shared_ptr<CompositeConcreteType>& composite_type_ptr) {
56 // Get the underlying object
57 CompositeConcreteType& composite_type = *composite_type_ptr;
58
59 bool is_union = std::holds_alternative<UnionType>(composite_type);
60
61 struc_t* ida_struct =
62 get_struc(std::visit([](const auto& t) { return t.id; }, composite_type));
63
64 for (ea_t member_offset = get_struc_first_offset(ida_struct);
65 member_offset != BADADDR;
66 member_offset = get_struc_next_offset(ida_struct, member_offset)) {
67 member_t* ida_member = get_member(ida_struct, member_offset);
68 if (ida_member == nullptr)
69 continue;
70
71 asize_t size;
72 if (is_varmember(ida_member)) {
73 QLOGE << "Found variable member `"
74 << ConvertIdaString(get_member_name(ida_member->id))
75 << "` that has variable size! Use size of 0.";
76 size = 0;
77 }
78 size = get_member_size(ida_member);
79
80 tinfo_t tinf;
81 DataType type;
82 std::string base_type_name;
83 if (get_member_tinfo(&tinf, ida_member)) {
84 type = GetDataType(tinf);
85 // Get struct or union type name for later
86 qstring ida_string;
87 tinf.get_type_name(&ida_string);
88 base_type_name = ConvertIdaString(ida_string);
89 } else {
90 QLOGE << "Cannot get the tinfo_t * for member `"
91 << ConvertIdaString(get_member_name(ida_member->id));
92 type = TYPE_UNK;
93 }
94
95 // Unions always have 0 offsets
96 const ea_t offset = (is_union ? 0 : ida_member->soff);
97
98 // Emplace the CompositeTypeMember
99 auto& member = std::visit(
100 [&](auto& composite) -> auto& {
101 return composite.members.emplace_back(
102 offset, ConvertIdaString(get_member_name(ida_member->id)), type,
103 size);
104 },
105 composite_type);
106
107 // Add the composite type pointer if the member is composite as well
108 if (type == TYPE_STRUCT || type == TYPE_UNION) {
109 // Ask the CompositeTypes manager to give us the relevant struct/union
110 const auto& composite_types = CompositeTypes::GetInstance();
111 const auto& it = composite_types.get_by_name(base_type_name);
112
113 if (it == composite_types.end()) {
114 QLOGE << "Member `" << ConvertIdaString(get_member_name(ida_member->id))
115 << "` is of composite type `" << base_type_name
116 << "` but it was not found within the exported composite
117 types.";
118 } else {
119 member.composite_type_ptr = *it;
120 }
121 }
122
123 /* Retrieve comments */
124 std::visit(
125 [&](auto& composite) {
126 GetStructureMemberComment(composite_type_ptr,
127 composite.members.size(), ida_member->id);
128 },
129 composite_type);
130
131 // TODO references
132 // ExportStructureMemberReference(ea_t(ida_member->id),
133 // structure->members.back(),
134 // STRUCT_STRUCT);
135 }
136}

149static void ExportStructOrUnion(struc_t* ida_struct) {
150 if (ida_struct->is_varstr()) {
151 QLOGE << "Found composite type `"
152 << ConvertIdaString(get_struc_name(ida_struct->id))
153 << "` that has variable size! Ignoring it.";
154 return;
155 }
156
157 CompositeTypes& composite_types = CompositeTypes::GetInstance();
158 size_t size = get_struc_size(ida_struct);
159
160 if (is_union(ida_struct->id))
161 composite_types.emplace_back<UnionType>(
162 ConvertIdaString(get_struc_name(ida_struct->id)), ida_struct->id, size);
163 else
164 composite_types.emplace_back<StructureType>(
165 ConvertIdaString(get_struc_name(ida_struct->id)), ida_struct->id, size);
166
167 // TODO references
168 // ExportStructureReference(ea_t(structure->addr), structure,
169 // STRUCT_STRUCT);
170
171 // Print the type as a C-string if possible
172 {
173 qstring struc_name = get_struc_name(ida_struct->id);
174 tinfo_t tif;
175 if (tif.get_named_type(get_idati(), struc_name.c_str())) {
176 qstring decl;
177 if (tif.print(&decl, struc_name.c_str(),
178 PRTYPE_TYPE | PRTYPE_MULTI | PRTYPE_DEF | PRTYPE_SEMI))
179 std::visit([&](auto& t) { t.c_str = ConvertIdaString(decl); },
180 *composite_types.back());
181 }
182 }
183
184 GetCompositeTypeComment(composite_types.back());
185}

187void ExportCompositeDataTypes() {
188 // First export all the composite types
189 uval_t idx = get_first_struc_idx();
190
191 // FIX: Even if IDA SDK says the opposite, get_first_struc_idx() may
192 // return NULL when no structures_ are defined
193 while (idx != BADADDR) {
194 tid_t struct_idx = get_struc_by_idx(idx);
195 struc_t* ida_struct = get_struc(struct_idx);
196 if (ida_struct != nullptr)
197 ExportStructOrUnion(ida_struct);
198
199 idx = get_next_struc_idx(idx);
200 }
201
202 // After Exporting all the composite types, export the members of the
203 // composite types. We need to do it in this order or we won't be able to have
204 // members referencing structures that have yet to be exported or are
205 // incomplete.
206 // For ex: struct A { A *a; };
207 CompositeTypes& composite_types = CompositeTypes::GetInstance();
208 for (auto& composite_type_ptr : composite_types) {
209 std::visit(
210 [&composite_type_ptr](auto& composite) {
211 struc_t* ida_struct = get_struc(composite.id);
212
213 if (ida_struct->memqty != 0)
214 ExportCompositeMembers(composite_type_ptr);
215 },
216 *composite_type_ptr);
217 }
218}

229static void ExportEnumMembers(std::shared_ptr<EnumType>& enum_type,
230 enum_t ida_enum) {
231 class EnumMemberVisitor : public enum_member_visitor_t {
232 private:
233 std::shared_ptr<EnumType>& enum_type;
234
235 public:
236 EnumMemberVisitor(std::shared_ptr<EnumType>& enum_type)
237 : enum_type(enum_type) {}
238
239 int idaapi visit_enum_member(const_t cid, uval_t value) override {
240 qstring member_name;
241 get_enum_member_name(&member_name, cid);
242
243 enum_type->values.emplace_back(ConvertIdaString(member_name),
244 static_cast<int64_t>(value));
245
246 /* Retrieve comments */
247 GetEnumMemberComment(enum_type, enum_type->values.size(), cid);
248
249 // TODO references
250 // ExportStructureMemberReference(
251 // ea_t(cid), this->enumeration->members.back(), STRUCT_ENUM);
252 return 0;
253 }
254 };
255
256 EnumMemberVisitor visitor(enum_type);
257 for_all_enum_members(ida_enum, visitor);
258}

260void ExportEnums() {
261 auto& enums = Enums::GetInstance();
262
263 enum_t ida_enum;
264 for (int ida_enum_idx = 0; (ida_enum = getn_enum(ida_enum_idx)) != BADADDR;
265 ++ida_enum_idx) {
266 auto& enum_type =
267 enums.emplace_back(ConvertIdaString(get_enum_name(ida_enum)), ida_enum);
268
269 // Export the values (aka members)
270 if (get_enum_size(ida_enum) > 0)
271 ExportEnumMembers(enum_type, ida_enum);
272
273 // Print the enum as a C-string if possible
274 {
275 qstring name = get_enum_name(ida_enum);
276 tinfo_t tif;
277 if (tif.get_named_type(get_idati(), name.c_str())) {
278 qstring decl;
279 if (tif.print(&decl, name.c_str(),
280 PRTYPE_TYPE | PRTYPE_MULTI | PRTYPE_DEF | PRTYPE_SEMI))
281 enum_type->c_str = ConvertIdaString(decl);
282 }
283 }
284
285 // TODO References
286 // ExportStructureReference(ea_t(ida_enum), structure, STRUCT_STRUCT);
287
288 // Check for comment for the enum
289 GetEnumComment(enum_type);
290 }
