68constexpr BlockType RetrieveBlockType(fc_block_type_t block_type) {
69 switch (block_type) {
70 case fcb_normal:
71 return BTYPE_NORMAL;
72 case fcb_indjump:
73 return BTYPE_INDJUMP;
74 case fcb_ret:
75 return BTYPE_RET;
76 case fcb_cndret:
77 return BTYPE_CNDRET;
78 case fcb_noret:
79 return BTYPE_NORET;
80 case fcb_enoret:
81 return BTYPE_ENORET;
82 case fcb_extern:
83 return BTYPE_EXTERN;
84 case fcb_error:
85 return BTYPE_ERROR;
86 }
87 assert(false && "Invalid block type");
88}
