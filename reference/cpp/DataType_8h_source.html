253template <typename T>
254constexpr Quokka::CompositeType::CompositeSubType CompositeSubTypeToProto() {
255 using U = std::remove_cvref_t<T>;
256 if constexpr (std::is_same_v<U, UnionType>)
257 return Quokka_CompositeType_CompositeSubType_TYPE_UNION;
258 else if constexpr (std::is_same_v<U, StructureType>)
259 return Quokka_CompositeType_CompositeSubType_TYPE_STRUCT;
260 else if constexpr (std::is_same_v<U, PointerType>)
261 return Quokka_CompositeType_CompositeSubType_TYPE_POINTER;
262 else if constexpr (std::is_same_v<U, ArrayType>)
263 return Quokka_CompositeType_CompositeSubType_TYPE_ARRAY;
264 else if constexpr (std::is_same_v<U, TypedefType>)
265 return Quokka_CompositeType_CompositeSubType_TYPE_TYPEDEF;
266 else
267 static_assert(false, "Mismatch between the CompositeSubTypes");
268}

466template <typename F, typename T>
467concept visit_callback =
468 std::invocable<F&, T&> || std::invocable<F&, const type_uid_t&, T&>;
469
470template <typename... Ts, typename Variant, typename F>
471 requires((is_one_of_variant_v<Ts, Variant>) && ...)
472void visit_selected(Variant&& v, F&& fn) {
473 std::visit(
474 [&]<typename U>(U&& x) {
475 using X = std::remove_cvref_t<U>;
476 if constexpr ((std::same_as<X, Ts> || ...)) {
477 std::invoke(std::forward<F>(fn), std::forward<U>(x));
478 }
479 },
480 std::forward<Variant>(v));
481}
482
483template <typename... Ts, typename DT, typename F>
484 requires std::same_as<std::remove_cvref_t<DT>, DataTypes> &&
485 ((visit_callback<F, std::conditional_t<
486 std::is_const_v<std::remove_reference_t<DT>>,
487 const Ts, Ts>>) &&
488 ...)
489static inline void for_each_visit(DT&& dt, F&& fn) {
490 for (auto&& [tid, var] : dt) {
491 visit_selected<Ts...>(var, [&](auto&& x) {
492 if constexpr (std::invocable<F&, decltype(x)>) {
493 std::invoke(fn, std::forward<decltype(x)>(x));
494 } else if constexpr (std::invocable<F&, decltype((tid)), decltype(x)>) {
495 std::invoke(fn, tid, std::forward<decltype(x)>(x));
496 } else {
497 static_assert(always_false_v<decltype(x)>,
498 "for_each_visit: callback signature mismatch");
499 }
500 });
501 }
